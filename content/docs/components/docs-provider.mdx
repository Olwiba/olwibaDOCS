## Installation

```sh
npm install @olwiba/docs
```

## Usage

```tsx
import { DocsProvider } from '@olwiba/docs';
```

```tsx
const searchItems = [
  { label: 'Get Started', href: '/docs' },
  { label: 'Components', href: '/docs/components' },
];

function RootDocument({ children }) {
  return (
    <DocsProvider searchItems={searchItems}>
      {children}
    </DocsProvider>
  );
}
```
