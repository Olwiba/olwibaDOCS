## Import

```tsx
import { DocsLayout } from '@olwiba/docs';
```

## Usage

```tsx
import { DocsLayout } from '@olwiba/docs';

function DocsPage({ loaderData, pageTree, children }) {
  const sections = [
    { name: 'Get Started', href: '/docs' },
    { name: 'Components', href: '/docs/components' },
  ];

  return (
    <DocsLayout loaderData={loaderData} pageTree={pageTree} sections={sections}>
      {children}
    </DocsLayout>
  );
}
```
