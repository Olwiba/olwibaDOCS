## Installation

```sh
npm install @olwiba/docs
```

## Usage

```tsx
import { DocsSidebar, type SidebarSection } from '@olwiba/docs';
import { SidebarProvider } from '@olwiba/cn';
```

```tsx
const sections: SidebarSection[] = [
  { name: 'Get Started', href: '/docs' },
  { name: 'Components', href: '/docs/components' },
  { name: 'Themes', href: '/docs/themes' },
];

<SidebarProvider>
  <DocsSidebar tree={pageTree} sections={sections} />
  {/* Page content */}
</SidebarProvider>
```
