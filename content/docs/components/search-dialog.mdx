## Import

```tsx
import { SearchDialog, SearchButton } from '@olwiba/docs';
```

## Usage

### With RootProvider

The recommended way to use the search dialog is with the `RootProvider` from fumadocs-ui:

```tsx
import { RootProvider } from 'fumadocs-ui/provider/tanstack';
import { SearchDialog } from '@olwiba/docs';

function RootDocument({ children }) {
  return (
    <RootProvider
      search={{
        SearchDialog,
      }}
    >
      {children}
    </RootProvider>
  );
}
```

### SearchButton

Use the `SearchButton` component to trigger the search dialog:

```tsx
import { SearchButton } from '@olwiba/docs';

function Header() {
  return (
    <header>
      <SearchButton />
    </header>
  );
}
```

### Search API

You need to set up a search API endpoint for the dialog to work:

```ts
// routes/api/search/index.ts
import { createFileRoute } from '@tanstack/react-router';
import { createFromSource } from 'fumadocs-core/search/server';
import { source } from '~/lib/source';

const search = createFromSource(source, {
  language: 'english',
});

export const Route = createFileRoute('/api/search/')({
  server: {
    handlers: {
      GET: async ({ request }) => {
        return search.GET(request);
      },
    },
  },
});
```
