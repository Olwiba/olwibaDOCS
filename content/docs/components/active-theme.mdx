## Installation

```sh
npm install @olwiba/docs
```

## Usage

```tsx
import { ActiveThemeProvider, useThemeConfig, Theme } from '@olwiba/docs';
```

Wrap your app with `ActiveThemeProvider` (typically in the root layout):

```tsx
function RootDocument({ children }) {
  return (
    <ActiveThemeProvider>
      {children}
    </ActiveThemeProvider>
  );
}
```

Access and change the active theme from any component:

```tsx
function MyComponent() {
  const { activeTheme, setActiveTheme } = useThemeConfig();

  return (
    <button onClick={() => setActiveTheme(Theme.Blue)}>
      Current: {activeTheme}
    </button>
  );
}
```
