## Installation

```sh
npm install @olwiba/docs
```

## Usage

```tsx
import { APIReference } from '@olwiba/docs';
```

```mdx
<APIReference
  name="DocsSidebar"
  extends="Sidebar"
  props={[
    { name: "tree", type: "PageTree.Root", default: "-" },
    { name: "sections", type: "SidebarSection[]", default: "[]" },
  ]}
/>
```

## API Reference

<APIReference
  name="APIReference"
  props={[
    { name: "name", type: "string", default: "-" },
    { name: "extends", type: "string", default: "-" },
    { name: "props", type: "PropDef[]", default: "[]" },
  ]}
/>

```ts
interface PropDef {
  name: string;
  type: string;
  default?: string;
}
```
