## Installation

```sh
npm install @olwiba/docs
```

## Usage

```tsx
import { DocsHeader } from '@olwiba/docs';
```

```tsx
<DocsHeader
  logo="MyDocs"
  navItems={[
    { label: 'Docs', href: '/docs' },
    { label: 'Blog', href: '/blog' },
  ]}
  githubUrl="https://github.com/example/repo"
/>
```
