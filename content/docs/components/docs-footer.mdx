## Installation

```sh
npm install @olwiba/docs
```

## Usage

```tsx
import { DocsFooter } from '@olwiba/docs';
```

```tsx
// Default footer
<DocsFooter />

// Custom content
<DocsFooter>
  <p>My custom footer content</p>
</DocsFooter>
```
