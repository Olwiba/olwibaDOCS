## Introduction

`@olwiba/docs` provides everything you need to build beautiful documentation sites with TanStack Start and Fumadocs. It includes a blueprint-pattern layout, 7 color themes with runtime switching, mobile navigation, copy-page-for-AI functionality, enhanced code blocks, and API reference components.

## Installation

```bash
bun add @olwiba/docs @olwiba/cn fumadocs-core fumadocs-ui fumadocs-mdx
```

## Quick Setup

### 1. Configure Fumadocs MDX

Create a `source.config.ts` in your project root:

```ts title="source.config.ts"
import { defineConfig, defineDocs } from 'fumadocs-mdx/config';
import rehypePrettyCode from 'rehype-pretty-code';

export const docs = defineDocs({
  dir: 'content/docs',
});

export default defineConfig({
  mdxOptions: {
    rehypePlugins: (plugins) => {
      plugins.shift();
      plugins.push([
        rehypePrettyCode,
        {
          theme: {
            dark: 'github-dark',
            light: 'github-light-default',
          },
        },
      ]);
      return plugins;
    },
  },
});
```

### 2. Create Source Loader

```ts title="lib/source.ts"
import { createSource } from '@olwiba/docs/source';
import { docs } from 'fumadocs-mdx:collections/server';

export const source = createSource({
  source: docs.toFumadocsSource(),
  baseUrl: '/docs',
});
```

### 3. Update Vite Config

```ts title="vite.config.ts"
import mdx from 'fumadocs-mdx/vite';

export default defineConfig({
  plugins: [
    mdx(await import('./source.config')),
    // ... other plugins
  ],
});
```

### 4. Root Layout with ActiveThemeProvider

```tsx title="routes/__root.tsx"
import { ActiveThemeProvider } from '@olwiba/docs';
import { RootProvider } from 'fumadocs-ui/provider/tanstack';
import { SearchDialog } from '@olwiba/docs';

function RootDocument({ children }) {
  return (
    <html lang="en" suppressHydrationWarning>
      <body>
        <ActiveThemeProvider>
          <RootProvider search={{ SearchDialog }}>
            {children}
          </RootProvider>
        </ActiveThemeProvider>
      </body>
    </html>
  );
}
```

### 5. Docs Route

```tsx title="routes/docs/$.tsx"
import { DocsSidebar, DocsMobileNav, DocsToc, DocsCopyPage, mdxComponents } from '@olwiba/docs';
import { SidebarProvider } from '@olwiba/cn';

function Page() {
  return (
    <SidebarProvider>
      <DocsSidebar tree={pageTree} sections={sections} />
      <DocsMobileNav tree={pageTree} />
      <main>
        <DocsCopyPage page={rawContent} url={url} />
        <MDX components={{ ...defaultMdxComponents, ...mdxComponents }} />
        <DocsToc toc={toc} />
      </main>
    </SidebarProvider>
  );
}
```

### 6. Search API

```ts title="routes/api/search/index.ts"
import { createFromSource } from 'fumadocs-core/search/server';
import { source } from '~/lib/source';

const search = createFromSource(source, { language: 'english' });

export const Route = createFileRoute('/api/search/')({
  server: {
    handlers: {
      GET: async ({ request }) => search.GET(request),
    },
  },
});
```
